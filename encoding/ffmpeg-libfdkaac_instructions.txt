#
# Install prerequisites.
#
dnf install autoconf automake gcc gcc-c++ git libtool make nasm pkgconfig zlib-devel
dnf install yasm yasm-devel lame lame-devel

#
# libfdk-aac
#

# Get the latest fdk-aac audio encoder.
git clone git://github.com/mstorsjo/fdk-aac.git
cd fdk-aac

# Configure to disable shared lib and point to where ffmpeg will be built.
autoreconf -fiv
./configure --prefix="$HOME/ffmpeg_build" --disable-shared
make
make install

#
# ffmpeg
#

# Get using release tarball since there are actually major releases.
wget <http://ffmpeg.org...>
cd ffmpeg-*

# Point pkgconfig to the one in ffmpeg dir.
PKG_CONFIG_PATH="$HOME/ffmpeg_build/pkgconfig"
export PKG_CONFIG_PATH

# Configure. Not all these flags are necessary.
./configure --prefix="/usr/local/bin" --pkg-config-flags="--static" --extra-cflags="-I/usr/local/bin/include" --extra-ldflags="-L/usr/local/lib" --bindir="/usr/local/bin" --extra-libs="-ldl" --enable-gpl --enable-nonfree --enable-libfdk_aac
make
make install
